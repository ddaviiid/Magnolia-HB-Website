Helotypo — Self-Hosted 100% Match
Download from your Envato Elements subscription → place .woff2/.woff files in public/fonts/helotypo/ → use next/font/local. Two-minute setup, exact match.
// app/layout.tsx
import localFont from 'next/font/local'

const helotypo = localFont({
  src: [
    { path: '../public/fonts/helotypo/Helotypo-Light.woff2', weight: '300' },
    { path: '../public/fonts/helotypo/Helotypo-Regular.woff2', weight: '400' },
    { path: '../public/fonts/helotypo/Helotypo-Bold.woff2', weight: '700' },
  ],
  variable: '--font-helotypo'
})
